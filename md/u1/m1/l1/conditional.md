```py
# Define some variables
age = 25
name = "Alice"
score = 90
height = 1.75

# Equal operator
print(age == 25)  # Outputs: True

# Not equal operator
print(name != "Bob")  # Outputs: True

# Greater than operator
print(score > 80)  # Outputs: True

# Less than operator
print(height < 1.80)  # Outputs: True

# Greater than or equal to operator
print(score >= 90)  # Outputs: True

# Less than or equal to operator
print(height <= 1.75)  # Outputs: True
```
