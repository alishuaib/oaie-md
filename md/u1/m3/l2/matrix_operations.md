```py
matrix_A = np.array([[1, 2], [3, 4]])
matrix_B = np.array([[5, 6], [7, 8]])

# Addition
added_matrix = matrix_A + matrix_B
print("Matrix Addition:\n", added_matrix)

# Output:
# Matrix Addition:
# [[ 6  8]
#  [10 12]]

# Subtraction
subtracted_matrix = matrix_A - matrix_B
print("Matrix Subtraction:\n", subtracted_matrix)

# Output:
# Matrix Subtraction:
# [[-4 -4]
#  [-4 -4]]

# Element-wise multiplication
multiplied_matrix = matrix_A * matrix_B
print("Element-wise Multiplication:\n", multiplied_matrix)

# Output:
# Element-wise Multiplication:
# [[ 5 12]
#  [21 32]]
```
