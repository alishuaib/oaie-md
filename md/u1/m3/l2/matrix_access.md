```py
# Accessing element at row 1, column 2
element = matrix[1, 1]
print("Element at [1, 1]:", element)  # Output: Element at [1, 1]: 5

# Slicing rows or columns
first_row = matrix[0, :]  # Get all elements in the first row
second_column = matrix[:, 1]  # Get all elements in the second column
print("First row:", first_row)  # Output: First row: [1 2 3]
print("Second column:", second_column)  # Output: Second column: [4 5]
```
